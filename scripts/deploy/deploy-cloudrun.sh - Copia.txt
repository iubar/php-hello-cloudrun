#!/bin/bash

# === CONFIGURAZIONE ===
PROJECT_ID="tuo-project-id"
SERVICE_NAME="php-apache-service"
REGION="europe-west1"
IMAGE_NAME="php-apache-image"
PORT=8080

# === CLOUD SQL ===
CLOUDSQL_INSTANCE="project:region:instance"  # es: myproject:europe-west1:mydb

# === VARIABILI D‚ÄôAMBIENTE ===
ENV_VARS="APP_ENV=production,DEBUG=false"

# === SECRET MANAGER ===
# nome segreto: db-password, versione: latest
SECRET_ENV="DB_PASSWORD=projects/$PROJECT_ID/secrets/db-password/versions/latest"

# === LOGIN ===
echo "‚úÖ Autenticazione con Google Cloud..."
gcloud auth login

# === IMPOSTA IL PROGETTO ===
gcloud config set project $PROJECT_ID

# === COSTRUISCI IMMAGINE ===
echo "üê≥ Costruisco l'immagine Docker..."
gcloud builds submit --tag gcr.io/$PROJECT_ID/$IMAGE_NAME .

# === DEPLOY SU CLOUD RUN CON CLOUD SQL, ENV E SECRET ===
echo "üöÄ Deploy su Cloud Run con Cloud SQL, ENV e Secret Manager..."
gcloud run deploy $SERVICE_NAME \
  --image gcr.io/$PROJECT_ID/$IMAGE_NAME \
  --platform managed \
  --region $REGION \
  --allow-unauthenticated \
  --port $PORT \
  --add-cloudsql-instances $CLOUDSQL_INSTANCE \
  --set-env-vars $ENV_VARS \
  --set-secrets $SECRET_ENV

# === OUTPUT ===
echo "üéâ Deploy completato!"
gcloud run services describe $SERVICE_NAME \
  --platform managed \
  --region $REGION \
  --format="value(status.url)"
